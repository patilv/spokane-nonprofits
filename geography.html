Geography

Where the region’s nonprofits are — and where they aren’t

Key finding The region’s nonprofit infrastructure is heavily concentrated in two urban cores: Spokane County and Kootenai County together host about three-quarters of all active nonprofits in the seven-county region. But density tells a different story — the small university and rural counties (Whitman, Latah, Lincoln) actually have more nonprofits per resident than Spokane, while fast-growing Kootenai County has the region’s thinnest per-capita coverage.

Interactive map of nonprofit locations

Every dot is an active nonprofit at its registered (geocoded) address. Filter by category or county, and click any dot for details. Human services organizations (NTEE groups I, J, K, L, O, P) can be highlighted in red.

viewof mapCategory = Inputs.select(
  ["All categories"].concat(Array.from(new Set(orgs.map(d => d.subsector_name))).sort()),
  {label: "Category"}
)
viewof mapCounty = Inputs.select(
  ["All counties"].concat(Array.from(new Set(orgs.map(d => d.county))).sort()),
  {label: "County"}
)
viewof highlightHS = Inputs.toggle({label: "Highlight human services", value: true})
mappable = orgs.filter(d => d.lat != null && d.lon != null && !isNaN(d.lat))
mapFiltered = mappable.filter(d =>
  (mapCategory === "All categories" || d.subsector_name === mapCategory) &&
  (mapCounty === "All counties" || d.county === mapCounty)
)
dotMap = {
  const div = document.createElement("div");
  div.className = "map-frame";
  div.style.height = "540px";
  const map = L.map(div, {scrollWheelZoom: false});
  map.setView([47.55, -117.1], 7);
  L.tileLayer("https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}{r}.png", {
    attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> &copy; <a href="https://carto.com/attributions">CARTO</a>',
    maxZoom: 18
  }).addTo(map);
  const layer = L.layerGroup().addTo(map);
  const renderer = L.canvas({padding: 0.3});
  return {div, map, layer, renderer};
}
dotMap.div
mapUpdate = {
  const {map, layer, renderer, div} = dotMap;
  layer.clearLayers();
  // Leaflet needs the container to have been laid out before sizing correctly
  setTimeout(() => map.invalidateSize(), 60);
  const hsSet = new Set(["I", "J", "K", "L", "O", "P"]);
  for (const d of mapFiltered) {
    const isHS = highlightHS && hsSet.has(d.ntee_major);
    const m = L.circleMarker([d.lat, d.lon], {
      renderer,
      radius: isHS ? 4.5 : 3.5,
      color: "#ffffff",
      weight: 0.8,
      fillColor: isHS ? GU.red : GU.s1,
      fillOpacity: 0.75
    });
    m.bindPopup(
      `<strong>${d.name}</strong><br>` +
      `${d.city ?? ""}, ${d.state} ${d.zip}<br>` +
      `${d.subsector_name}${d.ntee ? " · NTEE " + d.ntee : ""}<br>` +
      (d.revenue > 0 ? `Most recent revenue: ${fmoney(d.revenue)}` : "No revenue reported")
    );
    layer.addLayer(m);
  }
  return mapFiltered.length;
}

Nonprofit density by ZIP code

Circle size shows the number of active nonprofits in each ZIP; color shows nonprofits per 1,000 residents (darker = denser). Downtown ZIPs act as the sector’s “headquarters districts” — nonprofit addresses cluster where offices, mail service, and visibility are, not necessarily where clients live.

zipMappable = zipGaps.filter(d => d.lat != null && !isNaN(d.lat) && d.population > 0)
zipColor = {
  const vals = zipMappable.map(d => d.orgs_per_1k).sort(d3.ascending);
  const scale = d3.scaleQuantile().domain(vals).range([GU.seq[0], GU.seq[2], GU.seq[4], GU.seq[5], GU.seq[6]]);
  return scale;
}
zipMap = {
  const div = document.createElement("div");
  div.className = "map-frame";
  div.style.height = "480px";
  const map = L.map(div, {scrollWheelZoom: false});
  map.setView([47.55, -117.1], 7);
  L.tileLayer("https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}{r}.png", {
    attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> &copy; <a href="https://carto.com/attributions">CARTO</a>',
    maxZoom: 18
  }).addTo(map);
  const rScale = d3.scaleSqrt().domain([1, d3.max(zipMappable, d => d.n_orgs)]).range([4, 26]);
  for (const d of zipMappable) {
    L.circleMarker([d.lat, d.lon], {
      radius: rScale(d.n_orgs),
      color: "#ffffff",
      weight: 1,
      fillColor: zipColor(d.orgs_per_1k),
      fillOpacity: 0.8
    }).bindPopup(
      `<strong>ZIP ${d.zip}</strong> · ${d.county}<br>` +
      `${fnum(d.n_orgs)} nonprofits · ${fnum(Math.round(d.population))} residents<br>` +
      `${d3.format(".1f")(d.orgs_per_1k)} per 1,000 residents<br>` +
      `${fnum(d.n_human_services)} human services orgs`
    ).addTo(map);
  }
  setTimeout(() => map.invalidateSize(), 60);
  return {div, map};
}

County comparison: totals vs. density

Raw counts put Spokane far ahead; per-capita density flips the ranking. High per-resident counts in Whitman, Latah, and Lincoln counties reflect how many civic institutions — granges, booster clubs, fraternal groups, college-town organizations — small communities sustain relative to their size.
